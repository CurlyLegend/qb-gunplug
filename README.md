# qb-gunplug

Simple Gang Weapon Crafting

-- Add this below to qb-core > shared > gangs.lua

    ['gunplug'] = {
		label = 'Gun Plug',
		grades = {
            ['0'] = {
                name = 'Gunrunner'
            },
            ['1'] = {
                name = 'The Plug',
                isboss = true
            },
        },
	},

Dependencies: 

qb-core
qb-target
qb-menu
